178void PPM::out_ppm(std::ostream& os) const {
179 os << (static_cast<int>(get_magic()) == 0 ? "P3" : "P6") << '\n';
180 os << get_width() << ' ' << get_height() << '\n';
181 os << get_max() << '\n';
182
183 if (get_max() <= 255) {
184 for (auto v : m_data) {
185 unsigned char b = static_cast<unsigned char>(v);
186 os.write(reinterpret_cast<const char*>(&b), 1);
187 }
188 } else {
189 for (auto v : m_data) {
190 // big-endian
191 unsigned char bytes[2]{
192 static_cast<unsigned char>((v >> 8) & 0xFF),
193 static_cast<unsigned char>(v & 0xFF)
194 };
195 os.write(reinterpret_cast<const char*>(bytes), 2);
196 }
197 }

200std::expected<PPM, PPM::Error> read_ppm(std::istream& is) {
201 std::string m{};
202 PPM::size_type w{};
203 PPM::size_type h{};
204 PPM::size_type max{};
205
206 if (!(is >> m) || (m != "P3" && m != "P6"))
207 return std::unexpected(PPM::Error{"Invalid magic number"});
208
209 if (!(is >> w) || w > PPM::MAX_WIDTH)
210 return std::unexpected(PPM::Error{"Width is invalid or out of range"});
211
212 if (!(is >> h) || h > PPM::MAX_HEIGHT)
213 return std::unexpected(PPM::Error{"Height is invalid or out of range"});
214
215 if (!(is >> max) || max > PPM::MAX_COLOR_VALUE)
216 return std::unexpected(PPM::Error{
217 "Max color value is invalid or out of range"
218 });
219
221 const PPM::size_type samples = w * h * 3;
222 PPM img{};
223 img.set_magic_num(m == "P3" ? PPM::MagicNum::P3 : PPM::MagicNum::P6);
224 img.set_width(w);
225 img.set_height(h);
226 img.set_max(max);
227 img.clear();
228
229 if (m != "P6") {
230 int v;
231
232 for (PPM::size_type i{}; i < samples; ++i) {
233 if (!(is >> v))
234 return std::unexpected(PPM::Error{"Unexpected EOF in P3 data"});
235 if (v > max)
236 return std::unexpected(PPM::Error{"Color value out of range"});
237 img.push_back(static_cast<PPM::data_type>(v));
238 }
239
240 return img;
241 }
242
243 is.ignore(255, '\n');
244
245 if (max <= 255) {
246 // 8-bit samples
247 for (PPM::size_type i{}; i < samples; ++i) {
248 std::uint8_t b = 0;
249 if (!is.read(reinterpret_cast<char*>(&b), 1))
250 return std::unexpected(PPM::Error{"Unexpected EOF in P6 data"});
251 img.push_back(static_cast<PPM::data_type>(b));
252 }
253 }
254
255 if (max > 255) {
256 // 16-bit samples
257 for (PPM::size_type i{}; i < samples; ++i) {
258 std::uint8_t hi = 0;
259 std::uint8_t lo = 0;
260 if (!is.read(reinterpret_cast<char*>(&hi), 1) ||
261 !is.read(reinterpret_cast<char*>(&lo), 1))
262 return std::unexpected(PPM::Error{"Unexpected EOF in P6 data"});
263
265 std::uint16_t v =
266 (static_cast<std::uint16_t>(hi) << 8) |
267 static_cast<std::uint16_t>(lo);
268
269 if (v > max)
270 return std::unexpected(PPM::Error{"Color value out of range"});
271
272 img.push_back(static_cast<PPM::data_type>(v));
273 }
274 }
275
276 return img;
277} // read_ppm
